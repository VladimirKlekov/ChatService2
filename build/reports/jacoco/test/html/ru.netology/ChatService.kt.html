package ru.netology

class ChatService {
    //Р”РµР»Р°СЋ СЃРїРёСЃРѕРє РґР»СЏ С‡Р°С‚РѕРІ
   val chats = mutableListOf<Chat>()
    private var messageId = 0L
    private var chatId = 0L

    //35 Р°СЂРјРёСЏ СЂРѕСЃСЃРёРё РІ СѓРєСЂР°РёРЅРµ

    fun deleteMessage(messageId: Int):Chat {

        return chats.removeAt(messageId)

    }

    fun readMessage(messageId: Int): List<Chat> {
        val chatCopy = chats.get(messageId)
           chats[messageId] = chatCopy .copy(
            readMessage = true)

        return chats

    }

    //P.S С„СѓРЅРєС†РёРё РЅР°Р·С‹РІР°СЋС‚СЃСЏ С„СѓРЅРєС†РёСЏРјРё РѕР±Р»Р°СЃС‚Рё РІРёРґРёРјРѕСЃС‚Рё (Р°РЅРіР». scope functions). Р’СЃРµРіРѕ РёС… РїСЏС‚СЊ: let, run, with, apply, Рё also.

    //С„СѓРЅРєС†РёСЏ РґР»СЏ СЃРѕР·РґР°РЅРёСЏ СЃРѕРѕР±С‰РµРЅРёСЏ
    fun createMessage(userId: Long, message: String, chatId: Long = 0): Message {
        //СЃРїРёСЃРѕРє С‡Р°С‚РѕРІ СЃ РёРЅРґРµРєСЃРѕРј id = id С‡Р°С‚Р°
        //IndexOfFirst() - РІРѕР·РІСЂР°С‰Р°РµС‚ РёРЅРґРµРєСЃ РїРµСЂРІРѕРіРѕ СЌР»РµРјРµРЅС‚Р°, СЃРѕРѕС‚РІРµС‚СЃС‚РІСѓСЋС‰РµРіРѕ Р·Р°РґР°РЅРЅРѕРјСѓ РїСЂРµРґРёРєР°С‚Сѓ РёР»Рё -1, РµСЃР»Рё С‚Р°РєРёС… СЌР»РµРјРµРЅС‚РѕРІ РЅРµС‚.
        val chatIndex = chats.indexOfFirst { it.id == chatId }

        //РџСЂРё РІС‹Р·РѕРІРµ takeIf РґР»СЏ РѕР±СЉРµРєС‚Р° СЃ РїСЂРµРґРёРєР°С‚РѕРј СЌС‚РѕС‚ РѕР±СЉРµРєС‚ Р±СѓРґРµС‚ РІРѕР·РІСЂР°С‰РµРЅ,
        //РµСЃР»Рё РѕРЅ СЃРѕРѕС‚РІРµС‚СЃС‚РІСѓРµС‚ РїСЂРµРґРёРєР°С‚Сѓ. Р’ РїСЂРѕС‚РёРІРЅРѕРј СЃР»СѓС‡Р°Рµ РІРѕР·РІСЂР°С‰Р°РµС‚СЃСЏ null
            .takeIf { it >= 0 }

            //run РљРѕРЅС‚РµРєСЃС‚РЅС‹Р№ РѕР±СЉРµРєС‚ РґРѕСЃС‚СѓРїРµРЅ РІ РєР°С‡РµСЃС‚РІРµ РїРѕР»СѓС‡Р°С‚РµР»СЏ (this). Р’РѕР·РІСЂР°С‰Р°РµРјРѕРµ Р·РЅР°С‡РµРЅРёРµ - СЂРµР·СѓР»СЊС‚Р°С‚ РІС‹РїРѕР»РЅРµРЅРёСЏ Р»СЏРјР±РґС‹.
            ?: run {

                //List.lastOrNull() РІРѕР·РІСЂР°С‰Р°РµС‚ РїРѕСЃР»РµРґРЅРёР№ СЌР»РµРјРµРЅС‚ РёР»Рё null, РµСЃР»Рё СЃРїРёСЃРѕРє РїСѓСЃС‚.
                chats.add(Chat(chats.lastOrNull()?.id ?: chatId, emptyList()))
                chats.lastIndex
            }

        val currentChat = chats[chatIndex]
        val newMessage = Message(authorId = userId, id = messageId++, text = message)
        chats[chatIndex] = currentChat.copy(
            messages = currentChat.messages + newMessage
        )

        return newMessage
    }

    fun getMessages(chatId: Long, offset: Int, startFrom: Int): List<Message> =
//Р¤СѓРЅРєС†РёСЏ single() РІРѕР·РІСЂР°С‰Р°РµС‚ РµРґРёРЅСЃС‚РІРµРЅРЅС‹Р№ СЌР»РµРјРµРЅС‚ РїРѕС‚РѕРєР°, РµСЃР»Рё РїРѕС‚РѕРє СЃРѕРґРµСЂР¶РёС‚ С‚РѕР»СЊРєРѕ РѕРґРёРЅ СЌР»РµРјРµРЅС‚.
        // Р•СЃР»Рё РїРѕС‚РѕРє РЅРµ СЃРѕРґРµСЂР¶РёС‚ СЌР»РµРјРµРЅС‚РѕРІ РіРµРЅРµСЂРёСЂСѓРµС‚СЃСЏ РёСЃРєР»СЋС‡РµРЅРёРµ NoSuchElementException, Р° РµСЃР»Рё
        // РІ РїРѕС‚РѕРєРµ Р±РѕР»СЊС€Рµ РѕРґРЅРѕРіРѕ СЌР»РµРјРµРЅС‚Р° - РёСЃРєР»СЋС‡РµРЅРёРµ IllegalStateException.
        chats.singleOrNull { it.id == chatId }
                //let
            //РљРѕРЅС‚РµРєСЃС‚РЅС‹Р№ РѕР±СЉРµРєС‚ РґРѕСЃС‚СѓРїРµРЅ РІ РєР°С‡РµСЃС‚РІРµ Р°СЂРіСѓРјРµРЅС‚Р° (it). Р’РѕР·РІСЂР°С‰Р°РµРјРѕРµ Р·РЅР°С‡РµРЅРёРµ - СЂРµР·СѓР»СЊС‚Р°С‚ РІС‹РїРѕР»РЅРµРЅРёСЏ Р»СЏРјР±РґС‹.
            .let { it?.messages ?: throw ChatNotFoundError() }
//РњРµС‚РѕРґ asSequence РєРѕР»Р»РµРєС†РёР№
//РјРѕР¶РЅРѕ СЃРѕР·РґР°С‚СЊ РїРѕСЃР»РµРґРѕРІР°С‚РµР»СЊРЅРѕСЃС‚СЊ РёР· РѕР±СЉРµРєС‚Р° Iterable (РЅР°РїСЂРёРјРµСЂ, РёР· РѕР±СЉРµРєС‚РѕРІ С‚РёРїР° List РёР»Рё Set)
            .asSequence()
//Р”Р»СЏ РїРѕР»СѓС‡РµРЅРёСЏ РІСЃРµС… СЌР»РµРјРµРЅС‚РѕРІ, РєСЂРѕРјРµ СѓРєР°Р·Р°РЅРЅРѕРіРѕ
// РєРѕР»РёС‡РµСЃС‚РІР° РїРµСЂРІС‹С… РёР»Рё РїРѕСЃР»РµРґРЅРёС… СЌР»РµРјРµРЅС‚РѕРІ, РёСЃРїРѕР»СЊР·СѓР№С‚Рµ С„СѓРЅРєС†РёРё drop() Рё dropLast()
            .drop(startFrom)
                //Р”Р»СЏ РїРѕР»СѓС‡РµРЅРёСЏ РѕРїСЂРµРґРµР»С‘РЅРЅРѕРіРѕ РєРѕР»РёС‡РµСЃС‚РІР° СЌР»РµРјРµРЅС‚РѕРІ, РЅР°С…РѕРґСЏС‰РёС…СЃСЏ РІ РЅР°С‡Р°Р»Рµ РєРѕР»Р»РµРєС†РёРё
            // (РЅР°С‡РёРЅР°СЏ СЃ РїРµСЂРІРѕРіРѕ Рё РґР°Р»РµРµ), РёСЃРїРѕР»СЊР·СѓР№С‚Рµ С„СѓРЅРєС†РёСЋ take(). Р’ СЃРІРѕСЋ РѕС‡РµСЂРµРґСЊ С„СѓРЅРєС†РёСЏ takeLast() РІРµСЂРЅС‘С‚ СЃРїРёСЃРѕРє
            .take(offset)
//ifEmpty Р’РѕР·РІСЂР°С‰Р°РµС‚ СЌС‚РѕС‚ РјР°СЃСЃРёРІ, РµСЃР»Рё РѕРЅ РЅРµ РїСѓСЃС‚, РёР»Рё СЂРµР·СѓР»СЊС‚Р°С‚ РІС‹Р·РѕРІР° С„СѓРЅРєС†РёРё defaultValue, РµСЃР»Рё РјР°СЃСЃРёРІ РїСѓСЃС‚.
            .ifEmpty { throw MessagesNotFoundError() }
                //РІРµСЂРЅСѓС‚СЊ СЃРїРёСЃРѕРє
            .toList()
}
